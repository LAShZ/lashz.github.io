func findRepeatNumber(nums []int) int {

            return val

            stat[val] = true

func findRepeatNumber(nums []int) int {
    n := len(nums)
    temp := make([]int, n)
    for i := 0; i<n;i++{
        temp[nums[i]]++
        if temp[nums[i]] > 1{
            return nums[i]
        }
    }
    return -1
}
