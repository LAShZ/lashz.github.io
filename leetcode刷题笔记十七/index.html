func isMatch(s string, p string) bool {
	slen := len(s)
	plen := len(p)
	dp := make([][]bool, slen+1)
	for i := 0; i <= slen; i++ {
		dp[i] = make([]bool, plen+1)
	}

	for i := 0; i <= slen; i++ {
		for j := 0; j < plen+1; j++ {
			if j == 0 {
				dp[i][j] = (i == 0)
			} else {
				if p[j-1] != '*' {
					if i > 0 && (s[i-1] == p[j-1] || p[j-1] == '.') {
						dp[i][j] = dp[i-1][j-1]
					}
				} else {
					if j >= 2 {
						dp[i][j] = dp[i][j-2] || dp[i][j]
					}
					if i >= 1 && j >= 2 && (s[i-1] == p[j-2] || p[j-2] == '.') {
						dp[i][j] = dp[i-1][j] || dp[i][j]
					}
				}
			}
		}
	}
	return dp[slen][plen]
}
