/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reversePrint(head *ListNode) []int {
    var length int
    temp := head
    for ;temp != nil; length++ {
        temp = temp.Next
    }
    list := make([]int, length)
    for ;head != nil; length-- {
        list[length - 1] = head.Val
        head = head.Next
    }
    return list
}
