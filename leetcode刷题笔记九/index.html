func minArray(numbers []int) int {
    if len(numbers) == 0 {
        return -1
    }
    
    for seq:= 1; seq < len(numbers); seq++ {
       if numbers[seq] < numbers[seq-1]{
           return numbers[seq]
       }
    }
    return numbers[0]
}

func minArray(numbers []int) int {
	low := 0
	high := len(numbers) - 1
	for low < high {
		pivot := low + (high - low) / 2
        if numbers[pivot] < numbers[high] {
            high = pivot
        } else if numbers[pivot] > numbers[high] {
            low = pivot + 1
        } else {
            high--
        }
	}
	return numbers[low]
}
